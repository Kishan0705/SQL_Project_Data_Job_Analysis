## üîç Question 5: What skills are required for top-paying Data Analyst Jobs?

```python

import pandas as pd
import matplotlib.pyplot as plt
import numpy as np

# Load your CSV file
path = r"path"
df = pd.read_csv(path)

# Process the skills column to count occurrences (same as before)
skills_series = df['skills'].str.split(',', expand=True).stack()
skills_series = skills_series.str.strip()
skills_df = skills_series.to_frame(name='skill')
skills_count = skills_df['skill'].value_counts().reset_index()
skills_count.columns = ['skills_name', 'count']

# Get the top 10 skills
top_skills = skills_count.head(10)

# Plotting the horizontal bar chart
plt.figure(figsize=(10, 6))
plt.style.use('dark_background')

# Choose colors for the bars
colors = plt.cm.Blues(np.linspace(0.3, 1, len(top_skills)))

# Create the horizontal bar chart
top_skills.plot(kind='barh', x='skills_name', y='count', color=colors, edgecolor='white', legend=False)

# Adding titles and labels
plt.title('Top 10 Most Demanded Data Analyst Skills for High-Paying Jobs', fontsize=16, color='white', pad=20)
plt.xlabel('Total Jobs', fontsize=12, color='white', labelpad=10)
plt.ylabel('Skills', fontsize=12, color='white', labelpad=10)

# Invert y-axis to show the most frequent skill at the top
plt.gca().invert_yaxis()

# Add light gridlines for readability
plt.grid(visible=True, axis='x', color='gray', linestyle='--', linewidth=0.5)

# Improve layout and appearance
plt.tight_layout()

# Display the chart
plt.show()

```
